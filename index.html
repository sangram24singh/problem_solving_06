Find the Smaller Angle

var Minimal_Angle = (h, m) => 
{
       h=h%12;
       let hour = (h*360)/12 + (m*360)/(12*60);
       let min = (m*360)/60;
       let angle = Math.abs(hour-min);
       if (angle>180)
       return 360 - angle;
       else 
       return angle;
};

Reverse a Number

var Reverse_Number = (N) => 
            {
              let rem,rev=0;
              while(N>0)
              {
                rem=N%10;
                rev=rev*10+rem;
                N=Math.floor(N/10);
              }
            
            return rev;
            };
